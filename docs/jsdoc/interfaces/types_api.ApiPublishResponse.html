    Interface ApiPublishResponse

    Publish response from Commands.com publish API

    Response structure returned after successfully publishing a stack to Commands.com. Contains the published stack's metadata and access information for verification and sharing.

    Example 

    const publishResponse: ApiPublishResponse = {
      org: 'developer123',
      name: 'awesome-tools',
      organizationUsername: 'developer123',
      url: 'https://commands.com/developer123/awesome-tools',
      version: '1.0.0',
      stack_id: 'st_abc123xyz789'
    };
