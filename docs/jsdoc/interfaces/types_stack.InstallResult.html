    Interface InstallResult

    Result of a stack installation operation

    Contains the outcome and details from installing a stack locally. Provides information about installed files and location for verification and cleanup purposes.

    Example 

    const result: InstallResult = {
      success: true,
      stackName: 'typescript-tools',
      installedFiles: [
        'package.json',
        'tsconfig.json',
        'src/index.ts'
      ],
      targetPath: '/Users/dev/my-project'
    };
